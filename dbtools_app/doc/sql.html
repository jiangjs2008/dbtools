1. 环境设定表

	Create TABLE appenv (
		[entrykey]   text PRIMARY KEY,
		[entryvalue] text
	);

2. 数据库驱动信息表

	Create TABLE driverinfo (
		[driverid]    smallint PRIMARY KEY,
		[drivertype]  text     UNIQUE NOT NULL,
		[description] text,
		[action]      text     NOT NULL,
		[driver]      text,
		[sampleurl]   text
	);

3. 常用数据库信息表

	Create TABLE favrinfo (
		[id]          smallint PRIMARY KEY,
		[driverid]    smallint NOT NULL,
		[name]        text     UNIQUE NOT NULL,
		[description] text,
		[url]         text     NOT NULL,
		[user]        text,
		[password]    text,
		[useflg]      tinyint  NOT NULL DEFAULT 1
	);

4. 画面文言一览表

	Create TABLE msglist (
		[msgid]    smallint PRIMARY KEY,
		[msgvalue] text     NOT NULL
	);
	msgid编排顺序：
	10000: 提示信息
	20000: 警告信息
	30000: 出错信息
	40000: 系统异常信息
